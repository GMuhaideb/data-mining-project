from statsmodels.tsa.arima.model import ARIMA
from statsmodels.graphics.tsaplots import plot_acf, plot_pacf
from statsmodels.tsa.stattools import adfuller

# Load the time series data into a Pandas DataFrame
M1 = pd.read_csv('Percentage change in the Industrial Production Index (IPI) by economic activity (monthly; change year-on-year)_data.csv', index_col='Time Period', parse_dates=True)

# Visualize the data
M1.plot()
plt.show()

# Check for stationarity
result = adfuller(M1['Percentage change in the Industrial Production Index'])
print('ADF Statistic: %f' % result[0])
print('p-value: %f' % result[1])

# Choose a model
model = ARIMA(M1['Percentage change in the Industrial Production Index'], order=(1, 1, 0))

# Fit the model
model_fit = model.fit()

# Evaluate the model
residuals = pd.DataFrame(model_fit.resid)
residuals.plot()
plt.show()
